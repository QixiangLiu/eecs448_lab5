create table Users{
user_id varchar(255) NOT NULL,
PRIMARY KEY (user_id)
};
create table Posts(
post_id int NOT NULL AUTO_INCREMENT,
content TEXT NOT NULL,
author_id varchar(255),
PRIMARY KEY (post_id),
FOREGIN KEY (author_id) REFERENCE Users(user_id)
);
